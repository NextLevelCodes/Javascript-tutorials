1.while Loop

Syntax:

        while (condition) {
            // code block
        }

Example:

        let i = 1;
        while (i <= 5) {
            console.log(i);
            i++;
        }

2.do...while Loop

Syntax:

        do {
            // code block
        } while (condition);

Example:

        let i = 1;
        do {
            console.log(i);
            i++;
        } while (i <= 5);

3.for Loop

Syntax:

        for (initialization; condition; update) {
            // code block
        }

Example:

        for (let i = 1; i <= 5; i++) {
            console.log(i);
        }
